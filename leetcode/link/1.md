    let curr = link;

      const v = curr.val;
      curr = curr.next;
      arr.push(v)
    }

      const v = arr[i - 1];
      const link = new ListNode(v);
      if (curr) {
        curr.next = link;
        curr = curr.next;
      } else {
        curr = link;
        head = link;
      }
      i--;
    }
    return head;

  const b = toNums(l2);
  const sum = a + b;
